cmake_minimum_required(VERSION 3.10)

project(CppAutoGUI)

set(CMAKE_CXX_STANDARD 11)

if(WIN32)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
    include_directories(${CMAKE_SOURCE_DIR}/cppautogui)
    add_executable(cppautogui cppautogui/cppautogui.cpp cppautogui/cppautogui_win.cpp)
elseif(APPLE)
    find_library(QUARTZ_FRAMEWORK Quartz)
    include_directories(${CMAKE_SOURCE_DIR}/cppautogui)
    add_executable(cppautogui cppautogui/cppautogui.cpp cppautogui/cppautogui_osx.cpp)
    target_link_libraries(cppautogui ${QUARTZ_FRAMEWORK})
elseif(UNIX)
    find_package(X11 REQUIRED)
    include_directories(${CMAKE_SOURCE_DIR}/cppautogui)
    add_executable(cppautogui cppautogui/cppautogui.cpp cppautogui/cppautogui_x11.cpp)
    target_link_libraries(cppautogui ${X11_LIBRARIES})
endif()

enable_testing()
add_executable(test_cppautogui tests/test_cppautogui.cpp)
add_test(NAME test_cppautogui COMMAND test_cppautogui)
